import pandas as pd
from matplotlib import pyplot as plt

data = [['Smooth Muscle', 0.147, 0.681], 
        ['Fibroblasts & Collagen', 0.825, 0.639],
        ['Adipose', 0.233, 0.032], 
      #  ['Inflammatory Aggregates', 0.041, 0.084], 
       # ['Arteries', 0.127, 0.065], 
        #['Veins & Lymphatics', 0.080, 0.155]
       ]
df = pd.DataFrame(data, columns = ['Tissue Structure', 
                                   'Correctly Classified as Normal (n=464)', 
                                   'Incorrectly Classified as Tumor (n=476)'
                                  ]
                 ) 
print(df)

df.plot(x = "Tissue Structure", 
        y = ['Correctly Classified as Normal (n=464)','Incorrectly Classified as Tumor (n=476)'],
        kind= 'bar',
        figsize=(10,7)
       )
plt.title("Normal Tissue Structure Analysis", fontsize=20, fontweight=1000)
plt.ylabel("Percentage of Tiles Containing the Histologic Feature", fontsize=14, rotation=90)
plt.xlabel(None)
plt.xticks(rotation=0, fontsize=14)
plt.legend(prop={'size':11.5})
plt.show()
